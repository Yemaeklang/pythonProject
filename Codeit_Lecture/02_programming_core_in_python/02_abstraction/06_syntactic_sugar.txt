# Syntactic Sugar :
# 자주 쓰이는 표현을 더 간략하게 쓸 수 있게 해주는 문법

# 다음 두 줄은 같습니다
    x = x + 1
    x += 1  # 덧셈 연산

# 다음 두 줄은 같습니다
    x = x + 2
    x += 2

# 다음 두 줄은 같습니다
    x = x * 2
    x *= 2  # 곱셈 연산

# 다음 두 줄은 같습니다
    x = x - 3
    x -= 3  # 뺄셈 연산

# 다음 두 줄은 같습니다
    x = x / 2
    x /= 2  # 나눗셈 연산

# 다음 두 줄은 같습니다
    x = x % 7
    x %= 7  # 나머지 연산 : x를 7로 나눈 나머지를 구하는 연산
